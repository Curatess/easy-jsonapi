def initialize(params = [])
  @params = {}
  params.each do |p|
    k = to_hash_key(p[0])
    @params[k] = Param.new(p[0], p[1])
  end
end

def remove(key)
  k = to_hash_key(key)
  deleted = @params.delete(k)
  deleted.value
end
