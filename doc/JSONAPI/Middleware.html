def call(env)

  req = nil
  if requesting_jsonapi_response?(env) || includes_jsonapi_document?(env)
    req = Rack::Request.new(env)
  end
  
  check_query_param_compliance(req)
  check_req_body_compliance(req, env)

  @app.call(env)
end
