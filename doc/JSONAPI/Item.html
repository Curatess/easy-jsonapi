def method_missing(method_name, *args, &block)
  return unless is_a? JSONAPI::Item
  if should_update_var?(method_name)
    @item[method_name[..-2].to_sym] = args[0]
  elsif should_get_var?(method_name)
    @item[method_name]
  else
    super
  end
end

def to_s
  tr = '{ '
  first = true
  @item.each do |k, v|
    if first
      first = false
      tr += "#{k} => '#{v}', "
    else
      tr += "#{k} => '#{v}'"
    end
  end
  tr += ' }'
end
